    2. Python Environment SetupÂ¶

    Option A: Using Poetry (Recommended)

    # Install Poetry if not already installed
curl -sSL https://install.python-poetry.org | python3 -

# Initialize Poetry project
poetry init

# Add Neuroglia and development dependencies
poetry add neuroglia-python[web]
poetry add --group dev pytest pytest-asyncio pytest-cov black flake8 mypy

# Create virtual environment and activate
poetry install
poetry shell

    Option B: Using venv

    # Create virtual environment
python -m venv venv

# Activate virtual environment
# On macOS/Linux:
source venv/bin/activate
# On Windows:
# venv\Scripts\activate

# Install dependencies
pip install neuroglia-python[web]
pip install pytest pytest-asyncio pytest-cov black flake8 mypy

    Application LoggingÂ¶

    src/config/logging.py:

    import logging
import sys
from pathlib import Path

def setup_logging(log_level: str = "INFO"):
    """Configure application logging"""

    # Create logs directory
    Path("logs").mkdir(exist_ok=True)

    # Configure logging
    logging.basicConfig(
        level=getattr(logging, log_level.upper()),
        format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
        handlers=[
            logging.FileHandler('logs/app.log'),
            logging.StreamHandler(sys.stdout)
        ]
    )

    # Configure third-party loggers
    logging.getLogger("uvicorn").setLevel(logging.INFO)
    logging.getLogger("fastapi").setLevel(logging.INFO)

    Create a validation script to ensure everything is set up correctly:

    scripts/validate-env.py:

    #!/usr/bin/env python3
"""Validate development environment setup"""

import sys
import subprocess
from pathlib import Path

def check_python_version():
    """Check Python version"""
    if sys.version_info < (3, 8):
        print("âŒ Python 3.8+ required")
        return False
    print(f"âœ… Python {sys.version_info.major}.{sys.version_info.minor}")
    return True

def check_dependencies():
    """Check if required packages are installed"""
    try:
        import neuroglia
        print("âœ… Neuroglia installed")
        return True
    except ImportError:
        print("âŒ Neuroglia not installed")
        return False

def check_docker():
    """Check if Docker services are running"""
    try:
        result = subprocess.run(
            ["docker", "ps"],
            capture_output=True,
            text=True,
            check=True
        )
        if "mongo" in result.stdout and "redis" in result.stdout:
            print("âœ… Docker services running")
            return True
        else:
            print("âš ï¸  Docker services not all running")
            return False
    except (subprocess.CalledProcessError, FileNotFoundError):
        print("âŒ Docker not available")
        return False

if __name__ == "__main__":
    checks = [
        check_python_version(),
        check_dependencies(),
        check_docker()
    ]

    if all(checks):
        print("\nğŸ‰ Development environment is ready!")
    else:
        print("\nâŒ Some issues need to be resolved")
        sys.exit(1)
